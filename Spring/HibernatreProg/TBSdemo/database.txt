SALARY FLOAT, BONUS INT, PAY_PER_HOUR FLOAT,

);  



CREATE TABLE emp122(  
ID  INT,NAME VARCHAR(255));  

CREATE TABLE regemp122(  
ID  INT,NAME VARCHAR(255), 
SALARY FLOAT, BONUS INT);  

CREATE TABLE contemp122(  
ID  INT,NAME VARCHAR(255), 
 PAY_PER_HOUR FLOAT,
CONTRACT_DURATION VARCHAR(255));  





CREATE TABLE emp123(  
ID  INT ,NAME VARCHAR(255),
PRIMARY KEY (ID)
); 

CREATE TABLE regemp123 (
    EID int,
    NAME VARCHAR(255),SALARY FLOAT,
    BONUS int,
    PRIMARY KEY (EID),
    FOREIGN KEY (EID) REFERENCES emp123(ID)
);


CREATE TABLE contemp123(  
EID  INT,NAME VARCHAR(255), 
 PAY_PER_HOUR FLOAT,
CONTRACT_DURATION VARCHAR(255),
PRIMARY KEY (EID),
    FOREIGN KEY (EID) REFERENCES emp123(ID));
